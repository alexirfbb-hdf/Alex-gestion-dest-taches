// IRFBB TASKS v4 — Apps Script // Déployer → Gérer les déploiements → Modifier → Nouvelle version const TASKS_SHEET_ID = '17boKs_ZsgxatiRZUD9XDVpNFU6aBfvxYYjBouMylzAA'; const TASKS_SHEET_NAME = 'Tâches'; const TASKS_COL = { id:1, liste:2, titre:3, notes:4, date_echeance:5, statut:6, priorite:7, created:8, recurrence:9, checklist:10, recurrence_detail:11 }; function doGet(e) { try { const ss = SpreadsheetApp.openById(TASKS_SHEET_ID); const sheet = ss.getSheetByName(TASKS_SHEET_NAME); if (!sheet) return jsonR({ error: 'Sheet introuvable' }); const data = sheet.getDataRange().getValues(); const headers = data[0].map(h => h.toString().trim()); const rows = []; for (let i = 1; i < data.length; i++) { if (!data[i][2]) continue; const obj = {}; headers.forEach((h, j) => obj[h] = (data[i][j] || '').toString().trim()); obj._row = i + 1; rows.push(obj); } return jsonR({ status: 'ok', tasks: rows }); } catch(err) { return jsonR({ error: err.toString() }); } } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(TASKS_SHEET_ID); const sheet = ss.getSheetByName(TASKS_SHEET_NAME); if (!sheet) return jsonR({ error: 'Sheet introuvable' }); if (data.action === 'update') { const row = parseInt(data.rowIndex); if (row < 2) return jsonR({ error: 'rowIndex invalide' }); const t = data.task; sheet.getRange(row, TASKS_COL.titre).setValue(t.titre||''); sheet.getRange(row, TASKS_COL.liste).setValue(t.liste||''); sheet.getRange(row, TASKS_COL.notes).setValue(t.notes||''); sheet.getRange(row, TASKS_COL.date_echeance).setValue(t.date_echeance||''); sheet.getRange(row, TASKS_COL.statut).setValue(t.statut||'active'); sheet.getRange(row, TASKS_COL.priorite).setValue(t.priorite||''); sheet.getRange(row, TASKS_COL.recurrence).setValue(t.recurrence||''); sheet.getRange(row, TASKS_COL.checklist).setValue(t.checklist||''); sheet.getRange(row, TASKS_COL.recurrence_detail).setValue(t.recurrence_detail||''); return jsonR({ status: 'updated', row }); } else if (data.action === 'append') { const t = data.task; const newRow = sheet.getLastRow() + 1; sheet.getRange(newRow, TASKS_COL.id).setValue(t.id||'local_'+Date.now()); sheet.getRange(newRow, TASKS_COL.liste).setValue(t.liste||''); sheet.getRange(newRow, TASKS_COL.titre).setValue(t.titre||''); sheet.getRange(newRow, TASKS_COL.notes).setValue(t.notes||''); sheet.getRange(newRow, TASKS_COL.date_echeance).setValue(t.date_echeance||''); sheet.getRange(newRow, TASKS_COL.statut).setValue('active'); sheet.getRange(newRow, TASKS_COL.priorite).setValue(t.priorite||''); sheet.getRange(newRow, TASKS_COL.created).setValue(new Date().toISOString().substring(0,10)); sheet.getRange(newRow, TASKS_COL.recurrence).setValue(t.recurrence||''); sheet.getRange(newRow, TASKS_COL.checklist).setValue(t.checklist||''); sheet.getRange(newRow, TASKS_COL.recurrence_detail).setValue(t.recurrence_detail||''); return jsonR({ status: 'appended', row: newRow }); } else if (data.action === 'delete') { const row = parseInt(data.rowIndex); if (row < 2) return jsonR({ error: 'rowIndex invalide' }); sheet.deleteRow(row); return jsonR({ status: 'deleted', row }); } else { return jsonR({ error: 'action inconnue' }); } } catch(err) { return jsonR({ error: err.toString() }); } } function jsonR(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
